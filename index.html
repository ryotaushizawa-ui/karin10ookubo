function startAccident() { const scene = document.getElementById('accident-scene'); const truck = document.getElementById('truck'); const txt = document.getElementById('accident-txt'); scene.style.display = 'flex'; scene.style.background = "#fff"; // 背景を白に戻す txt.innerText = "あ！危ない！！"; let pos = -250; // 速度を 45 から 5 に変更（数字が小さいほどゆっくりになります） const drive = setInterval(() => { pos += 5; truck.style.right = pos + 'px'; // 画面中央付近で衝突 if (pos > window.innerWidth / 2 - 100) { txt.innerText = "ガシャーーーーン！！"; scene.style.background = "red"; // 衝撃で赤く } // 画面外に消えたら停止 if (pos > window.innerWidth + 300) { clearInterval(drive); } }, 10); // トラックが通り過ぎるまで待ってからクイズ画面へ（時間を 1500ms から 3500ms に延長） setTimeout(() => { scene.style.display = 'none'; document.getElementById('start-screen').classList.add('hidden'); document.getElementById('quiz').classList.remove('hidden'); commentBox.innerText = "「ひどい事故ねぇ。でもそのおかげでアンタ、才能が開花しそうよ？」"; showQ(); }, 3500); }
